cmake_minimum_required(VERSION 3.15)
project(PasswordGenerator_C)

if(MSVC)
    add_compile_options(/utf-8)
endif()

option(BUILD_SHARED_LIBS "Build shared libraries" OFF)
add_library(password_generator_lib lib/password_generator.c lib/password_generator.h)
add_executable(generate app/main.c)
target_link_libraries(generate password_generator_lib)
target_include_directories(generate PRIVATE lib)
add_dependencies(generate password_generator_lib)
